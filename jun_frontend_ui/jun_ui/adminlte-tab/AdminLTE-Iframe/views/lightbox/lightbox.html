lightbox的用法

<!-- 1. 引入文件 -->
<script src="https://cdn.bootcdn.net/ajax/libs/jquery/3.5.0/jquery.min.js">

<link rel="stylesheet" href="https://cdn.bootcdn.net/ajax/libs/ekko-lightbox/5.3.0/ekko-lightbox.css" />
<script src="https://cdn.bootcdn.net/ajax/libs/ekko-lightbox/5.3.0/ekko-lightbox.js">

<!-- 2. 编写HTML -->
<a data-toggle="lightbox" data-title="你好，单图1" data-footer="你好，单图1" href="http://placehold.it/150x100">
  <img src="http://placehold.it/150x100" class="img-circle" style="width:100px;height:100px;">
</a>

<a data-toggle="lightbox" data-title="你好，多图1" data-footer="你好，多图1" data-gallery="gallery" href="http://placehold.it/150x100">
  <img src="http://placehold.it/150x100" class="img-circle" style="width:100px;height:100px;">
</a>

<!-- 3. 初始化 -->
<script>
  $(document).on('click', '[data-toggle="lightbox"]', function(event) {
    event.preventDefault();
    $(this).ekkoLightbox();
  });
</script>
